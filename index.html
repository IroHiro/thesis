""" TERRASIGHT PREDICTION API - Main Backend Run this on Render """ import numpy as np import joblib import os from flask import Flask, request, jsonify from flask_cors import CORS app = Flask(__name__) CORS(app) BASE_DIR = os.path.dirname(os.path.abspath(__file__)) print("=" * 60) print("TERRASIGHT PREDICTION API - Starting up...") print("=" * 60) # ============================================ # LOAD MODELS # ============================================ # Load Random Forest rf_path = os.path.join(BASE_DIR, 'random_forest_model.pkl') if os.path.exists(rf_path): rf_model = joblib.load(rf_path) print(f"✅ Random Forest loaded - expects {rf_model.n_features_in_} features") else: rf_model = None print(f"❌ random_forest_model.pkl not found") # Load encoders crop_encoder_path = os.path.join(BASE_DIR, 'crop_encoder.pkl') if os.path.exists(crop_encoder_path): crop_encoder = joblib.load(crop_encoder_path) print(f"✅ Crop encoder loaded: {crop_encoder.classes_}") else: crop_encoder = None season_encoder_path = os.path.join(BASE_DIR, 'season_encoder.pkl') if os.path.exists(season_encoder_path): season_encoder = joblib.load(season_encoder_path) print(f"✅ Season encoder loaded: {season_encoder.classes_}") else: season_encoder = None muni_encoder_path = os.path.join(BASE_DIR, 'municipality_encoder.pkl') if os.path.exists(muni_encoder_path): muni_encoder = joblib.load(muni_encoder_path) print(f"✅ Municipality encoder loaded: {muni_encoder.classes_}") else: muni_encoder = None print("=" * 60) # ============================================ # HELPER FUNCTIONS # ============================================ def process_weekly_sequence(raw_sequence): """Convert 4 weeks of weather data into a feature vector""" all_features = [] for week in raw_sequence: features = [ week[0], # ndvi week[1], # evi week[4], # temp_avg week[2], # temp_max week[3], # temp_min week[5], # rainfall week[6], # humidity week[7], # solar_radiation week[8] # wind_speed ] all_features.extend(features) # Calculate statistical features across all 4 weeks ndvi_vals = [w[0] for w in raw_sequence] evi_vals = [w[1] for w in raw_sequence] temp_vals = [w[4] for w in raw_sequence] rain_vals = [w[5] for w in raw_sequence] stats = [ np.mean(ndvi_vals), np.max(ndvi_vals), np.min(ndvi_vals), np.mean(evi_vals), np.max(evi_vals), np.min(evi_vals), np.mean(temp_vals), np.max(temp_vals), np.min(temp_vals), np.sum(rain_vals) ] return np.array(all_features + stats) def get_season_name(season_code): if season_code == 2: return 'Wet' elif season_code == 1: return 'Dry' return None def predict_yield(features, crop_name, municipality, season_name): """Predict yield using Random Forest""" if rf_model is None: return None, "Random Forest model not loaded" feature_list = list(features) # Add crop if crop_encoder: try: crop_val = crop_encoder.transform([crop_name])[0] except: crop_val = 1 if crop_name == 'Rice' else 0 else: crop_val = 1 if crop_name == 'Rice' else 0 feature_list.append(crop_val) # Add municipality if municipality and muni_encoder: try: if municipality in muni_encoder.classes_: muni_val = muni_encoder.transform([municipality])[0] feature_list.append(muni_val) except: pass # Add season if season_name and season_encoder: try: if season_name in season_encoder.classes_: season_val = season_encoder.transform([season_name])[0] feature_list.append(season_val) except: pass # Pad to match model expectation expected = rf_model.n_features_in_ if len(feature_list) < expected: feature_list.extend([0] * (expected - len(feature_list))) elif len(feature_list) > expected: feature_list = feature_list[:expected] features_array = np.array(feature_list).reshape(1, -1) predicted = rf_model.predict(features_array)[0] return predicted, None # ============================================ # API ENDPOINTS # ============================================ @app.route('/predict', methods=['POST']) def predict(): try: data = request.get_json() if not data: return jsonify({'error': 'No JSON data'}), 400 raw_sequence = data.get('raw_sequence') crop_encoded = data.get('crop_encoded') municipality = data.get('municipality') season_code = data.get('season') if not raw_sequence: return jsonify({'error': 'Missing raw_sequence'}), 400 if crop_encoded is None: return jsonify({'error': 'Missing crop_encoded'}), 400 if not municipality: return jsonify({'error': 'Missing municipality'}), 400 features = process_weekly_sequence(raw_sequence) crop_name = 'Rice' if crop_encoded == 1 else 'Corn' season_name = get_season_name(season_code) # Debug logging print(f"\n📥 REQUEST: {municipality}, {crop_name}, {season_name}") print(f" Features shape: {features.shape}") print(f" First 5 features: {features[:5]}") predicted, error = predict_yield(features, crop_name, municipality, season_name) if error: return jsonify({'error': error}), 500 print(f" Predicted: {predicted:.3f} tons/ha") return jsonify({ 'yield': round(predicted, 3), 'crop': crop_name, 'municipality': municipality, 'season': season_name, 'status': 'success' }) except Exception as e: print(f"❌ Error: {e}") import traceback traceback.print_exc() return jsonify({'error': str(e)}), 500 @app.route('/model_info', methods=['GET']) def model_info(): """Get model information for debugging""" return jsonify({ 'rf_features_expected': rf_model.n_features_in_ if rf_model else None, 'crop_encoder_classes': crop_encoder.classes_.tolist() if crop_encoder else None, 'municipality_encoder_classes': muni_encoder.classes_.tolist() if muni_encoder else None, 'season_encoder_classes': season_encoder.classes_.tolist() if season_encoder else None }) @app.route('/health', methods=['GET']) def health(): return jsonify({'status': 'healthy'}) if __name__ == '__main__': port = int(os.environ.get('PORT', 5000)) app.run(host='0.0.0.0', port=port)
